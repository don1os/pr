 #include <iostream>
 #include <vector>
 #include <unordered_set>
 #include <limits>
 using namespace std;
 
 int predict(vector<int>& pages, vector<int>& frame, int n, int index)
 {
    int res = -1, farthest = index;
    for (int i = 0; i < frame.size(); i++)
    {
        int j;
        for (j = index; j < n; j++)
        {
            if (frame[i] == pages[j])
            {
                if (j > farthest)
                {
                    farthest = j;
                    res = i;
                }
                break;
            }
        }
        if (j == n) 
            return i;
    }
    return (res == -1) ? 0 : res;
 }
 
int main()
 {
    int n, capacity;
    cout << "Enter number of pages: ";
    cin >> n;
 
    vector<int> pages(n);
    cout << "Enter the page reference string: ";
    for (int i = 0; i < n; i++)
        cin >> pages[i];
 
    cout << "Enter frame capacity: ";
    cin >> capacity;
 
    vector<int> frame;
    int page_faults = 0, page_hits = 0;
 
    for (int i = 0; i < n; i++)
    {
    
        bool hit = false;
        for (int j = 0; j < frame.size(); j++)
        {
            if (frame[j] == pages[i])
            {
                hit = true;
                page_hits++;
                break;
            }
        }
 
        if (!hit)
        {
            page_faults++;
            if (frame.size() < capacity)
            {
                frame.push_back(pages[i]);
            }
            else
            {
                int j = predict(pages, frame, n, i + 1);
                frame[j] = pages[i];
            }
        }
    }
 
    cout << "\nTotal Page Faults: " << page_faults << endl;
    cout << "Total Page Hits: " << page_hits << endl;
    return 0;
 }
