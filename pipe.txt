#include <stdio.h>
#include <unistd.h>
#include <string.h>   
#include <sys/types.h>
#include <stdlib.h>  

int main() {
    int fd[2];

   
    if (pipe(fd) == -1) {
        perror("pipe");
        exit(1);
    }

    pid_t pid = fork();

    if (pid < 0) {
        perror("fork");
        exit(1);
    } 
    else if (pid > 0) {   
        close(fd[0]);     

        const char msg[] = "Hello from Parent!";
        write(fd[1], msg, strlen(msg) + 1);  

        close(fd[1]);   
    } 
    else {              
        close(fd[1]);    
        char buffer[50];
        ssize_t bytesRead = read(fd[0], buffer, sizeof(buffer));
        if (bytesRead > 0) {
            printf("Child received: %s\n", buffer);
        } else {
            perror("read");
        }

        close(fd[0]);     
    }

    return 0;
}
