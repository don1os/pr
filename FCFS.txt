#include <bits/stdc++.h>
 using namespace std;
 
void pageFaultsAndHits(int pages[], int n, int capacity)
 {
    unordered_set<int> s;   
    queue<int> indexes;     
 
    int page_faults = 0;
    int page_hits = 0;
 
    for (int i = 0; i < n; i++)
    {
        if (s.find(pages[i]) == s.end())   
        {
            if (s.size() < capacity)
            {
                s.insert(pages[i]);
                indexes.push(pages[i]);
                page_faults++;
            }
            else
            {
                int val = indexes.front();
                indexes.pop();
                s.erase(val);
 
                s.insert(pages[i]);
                indexes.push(pages[i]);
                page_faults++;
            }
        }
        else   
        {
            page_hits++;
        }
    }
 
    cout << "Total Page Faults: " << page_faults << endl;

    cout << "Total Page Hits: " << page_hits << endl;
}
 int main()
 {
 int n, capacity;
 cout << "Enter number of pages: ";
 cin >> n;
 int pages[n];
 cout << "Enter the page reference string: ";
 for (int i = 0; i < n; i++)
 cin >> pages[i];
 cout << "Enter frame capacity: ";
 cin >> capacity;
 pageFaultsAndHits(pages, n, capacity);
 return 0;
 }
